这里使用了shell脚本来实现对于make的编译，由于使用的是wsl在windows下操作导致编码格式含有\r\n，以此有两个shell脚本。

conf.sh

# 实现对于make的编译与运行
#!/bin/bash
if [[ $1 == clean ]]; then
        make clean
elif [[ $1 == all ]]; then
        make clean && make -j8 && ./main
else
        make -j8 && ./main

shell.sh

# 实现将conf.sh的dos编码转化为unix编码，并执行
#!/bin/bash
dos2unix  conf.sh && ./conf.sh

Makefile

# 自动捕捉符合c/c++编译
#指定编译工具
CC = gcc
CPP = g++
LINK = g++

LIBS = -lsqlite3 -lpthread
#编译.so 必须添加 -fPIC 和 -shared 选项
CCFLAGS = -c -g -fPIC
CPPFLAGS = -c -g -fPIC

#期望得到的执行文件或动态库.so
#TARGET=libxx.so
TARGET=main

INCLUDES = -I. #-I../../

CPPFILES = $(wildcard *.cpp )#遍历得到当前目录及上层目录中的所有.cpp文件
CFILES = $(wildcard *.c )#遍历得到当前目录及上层目录中的所有.c文件

OBJFILE = $(CFILES:.c=.o) $(CPPFILES:.cpp=.o)

all:$(TARGET)

$(TARGET):$(OBJFILE)
# 编译得到 .so 文件用下面的代码
#   $(LINK) $^ $(LIBS) -Wall -fPIC -shared -o $@
# 编译得到可执行文件用下面的代码
        $(LINK) $^ $(LIBS) -Wall -O2 -o $@

%.o:%.c
        $(CC) -o $@ $(CCFLAGS) $< $(INCLUDES)

%.o:%.cpp
        $(CPP) -o $@ $(CPPFLAGS) $< $(INCLUDES)

clean:
        rm -rf $(TARGET)
        rm -rf $(OBJFILE)

main.c

#include <stdio.h>

 
int main()
{
    printf("Hello, World! \n");
 
    return 0;
}

./shell
